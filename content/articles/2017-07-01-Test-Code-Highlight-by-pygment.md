求二叉树的第k个节点：

	public class KthNode
	{
		 public TreeNode kthNode(TreeNode pRoot, int k)
		 {
	        //求二叉排序树的第k个节点，只需要中序遍历，即可
	        if(pRoot == null || k < 1)
	            return null;
	        Stack<TreeNode> stack = new Stack<TreeNode>();
	        TreeNode p = pRoot;
	        TreeNode kthNode = null;
	        int count = 0;
	        while(p != null || !stack.empty() && count !=k ){
	            if(p != null){
	                stack.push(p);
	                p = p.left;
	            }
	            else{
	               p = stack.pop();
	               count++;
	               if(count == k)
	                   kthNode = p;
	               p = p.right;
	            }
	        }
	        return kthNode;
		 }
	}
